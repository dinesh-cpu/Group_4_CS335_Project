git commands :
 (1) git init // initialize git repo
  (2) git status  // get status
  (3) git add .  // add all file to staging area
  (4) git commit -m ""  // commit all staging area files
  (5) git stash  // send all staging file to back stash without commit
  (6) git stash pop  // pop from stash area
  (7) git reset "<hash of the commit>"  // go to perticular commit
  (8) git stash clear  // clear the stash area
  (9) git log   // get info about all commit
  (10) git remote add origin "<link>"  // add the origin
  (11) git remote -v  // check all remote links
  (12)git push origin master  // pushing to the origin link
  (13) git branch <branch name> //  to create the git branch
  (14) git checkout <branch name> // to change the branch
